```typescript
import { buildConfig } from 'payload'
import { postgresAdapter } from '@payloadcms/db-postgres'
import { lexicalEditor } from '@payloadcms/richtext-lexical'
import sharp from 'sharp'
import path from 'path'
import { fileURLToPath } from 'url'

import { Pages } from '@/collections/Pages'
import { RegulatoryUpdates } from '@/collections/RegulatoryUpdates'
import { Certifications } from '@/collections/Certifications'
import { NewsItems } from '@/collections/NewsItems'
import { Programmes } from '@/globals/Programmes'

const filename = fileURLToPath(import.meta.url)
const dirname = path.dirname(filename)

export default buildConfig({
  admin: {
    user: 'users',
    importMap: {
      baseDir: path.resolve(dirname, '..'),
      importMapFile: path.resolve(dirname, 'app/(payload)/admin/importMap.js'),
    },
  },
  collections: [
    {
      slug: 'users',
      auth: true,
      admin: { useAsTitle: 'email' },
      fields: [
        { name: 'name', type: 'text', required: true },
        {
          name: 'role',
          type: 'select',
          options: [
            { label: 'Admin', value: 'admin' },
            { label: 'Editor', value: 'editor' },
          ],
          defaultValue: 'editor',
          required: true,
        },
      ],
    },
    Pages,
    RegulatoryUpdates,
    Certifications,
    NewsItems,
  ],
  globals: [Programmes],
  editor: lexicalEditor(),
  secret: process.env.PAYLOAD_SECRET || '',
  typescript: {
    outputFile: path.resolve(dirname, 'payload-types.ts'),
  },
  db: postgresAdapter({
    pool: {
      connectionString: process.env.DATABASE_URI || '',
    },
  }),
  sharp,
})
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Admin panel -> Database | Editor input through Payload admin forms crosses into database storage |
| CMS data -> Frontend render | Data from database rendered on public frontend pages |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-01 | Spoofing | Admin panel access | mitigate | Payload's built-in auth (from Phase 1) requires email/password login. No changes needed -- existing auth protects all new collections. |
| T-02-02 | Tampering | Collection field values | mitigate | Payload validates field types server-side (required fields, select options, etc.). All select fields use fixed option lists -- editors cannot inject arbitrary values for category/issuingBody/pageType. |
| T-02-03 | Information Disclosure | Version history | accept | Version history is only visible to authenticated admin panel users. Low risk -- editors are trusted users. |
| T-02-04 | Injection | Rich text body field | mitigate | Payload's Lexical editor sanitizes rich text output. The richText field type produces structured JSON, not raw HTML. Frontend must use Payload's `<RichText />` component or serialize safely. |
| T-02-05 | Tampering | Slug field (URL injection) | mitigate | Slug field is a simple text field with unique constraint. Frontend uses `notFound()` for non-matching slugs. Validate slug format in collection hook or rely on Payload's text sanitization. |
</threat_model>
