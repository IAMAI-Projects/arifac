**1. Create `src/components/BlockRenderer.tsx`:**

A server component that maps block types to components. Uses a switch on `block.blockType` to render the correct component. Import the `Page` type from `@/payload-types`. The block union type is `NonNullable&lt;Page['layout']&gt;[number]`.

```typescript
import type { Page, RegulatoryUpdate } from '@/payload-types'
import Hero from '@/components/Hero'
import StatsStrip from '@/components/StatsStrip'
import CapabilityMatrix from '@/components/CapabilityMatrix'
import RegulatoryDashboard from '@/components/RegulatoryDashboard'
import FeaturedPrograms from '@/components/FeaturedPrograms'
import CTASection from '@/components/CTASection'

type Block = NonNullable<Page['layout']>[number]

interface BlockRendererProps {
  blocks: Block[]
  regulatoryUpdates?: RegulatoryUpdate[]
}

export default function BlockRenderer({ blocks, regulatoryUpdates = [] }: BlockRendererProps) {
  return (
    <>
      {blocks.map((block, index) => {
        switch (block.blockType) {
          case 'hero':
            return <Hero key={index} data={block} />
          case 'stats':
            return <StatsStrip key={index} data={block} />
          case 'capabilityMatrix':
            return <CapabilityMatrix key={index} data={block} />
          case 'regulatoryDashboard':
            return <RegulatoryDashboard key={index} data={block} updates={regulatoryUpdates} />
          case 'featuredPrograms':
            return <FeaturedPrograms key={index} data={block} />
          case 'cta':
            return <CTASection key={index} data={block} />
          default:
            return null
        }
      })}
    </>
  )
}
```

The BlockRenderer accepts an optional `regulatoryUpdates` prop because the RegulatoryDashboard block needs live update entries from the RegulatoryUpdates collection (not stored in the block itself -- the block only stores section chrome).

**2. Rewrite `src/app/(frontend)/page.tsx`:**

Remove `"use client"` directive. Make it an async server component. Fetch the home page document (slug: 'home') and regulatory updates via Payload Local API.

```typescript
import configPromise from '@payload-config'
import { getPayload } from 'payload'
import { notFound } from 'next/navigation'
import BlockRenderer from '@/components/BlockRenderer'

export default async function HomePage() {
  const payload = await getPayload({ config: configPromise })

  const result = await payload.find({
    collection: 'pages',
    where: { slug: { equals: 'home' } },
    limit: 1,
  })

  const page = result.docs[0]
  if (!page) notFound()

  // Fetch latest regulatory updates for the dashboard block
  const updatesResult = await payload.find({
    collection: 'regulatory-updates',
    sort: '-date',
    limit: 5,
  })

  return (
    <>
      {page.layout && page.layout.length > 0 && (
        <BlockRenderer
          blocks={page.layout}
          regulatoryUpdates={updatesResult.docs}
        />
      )}
    </>
  )
}
```
