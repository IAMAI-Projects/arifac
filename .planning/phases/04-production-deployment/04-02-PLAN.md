```hcl
resource "aws_ecr_repository" "app" {
  name                 = "${var.project_name}-web"
  image_tag_mutability = "MUTABLE"

  image_scanning_configuration {
    scan_on_push = true
  }
}

resource "aws_ecr_lifecycle_policy" "app" {
  repository = aws_ecr_repository.app.name

  policy = jsonencode({
    rules = [{
      rulePriority = 1
      description  = "Keep last 5 images"
      selection = {
        tagStatus   = "any"
        countType   = "imageCountMoreThan"
        countNumber = 5
      }
      action = { type = "expire" }
    }]
  })
}
```

4. **infra/ecs.tf** -- ECS cluster and service using terraform-aws-modules. Because the exact ECS Express Mode submodule API is an assumption (A1 in research), use the well-documented `terraform-aws-modules/ecs/aws` cluster module plus a standard Fargate service definition as a reliable baseline. The Express Mode can be swapped in once the user verifies the module API.

```hcl
module "ecs_cluster" {
  source  = "terraform-aws-modules/ecs/aws//modules/cluster"
  version = "~> 5.11"

  cluster_name = "${var.project_name}-cluster"

  # Fargate capacity providers
  fargate_capacity_providers = {
    FARGATE = {
      default_capacity_provider_strategy = {
        weight = 100
      }
    }
  }
}

resource "aws_ecs_task_definition" "app" {
  family                   = "${var.project_name}-web"
  requires_compatibilities = ["FARGATE"]
  network_mode             = "awsvpc"
  cpu                      = var.ecs_cpu
  memory                   = var.ecs_memory
  execution_role_arn       = aws_iam_role.ecs_execution.arn
  task_role_arn            = aws_iam_role.ecs_task.arn

  container_definitions = jsonencode([{
    name  = "app"
    image = "${aws_ecr_repository.app.repository_url}:latest"
    portMappings = [{
      containerPort = 3000
      protocol      = "tcp"
    }]
    environment = [
      { name = "NODE_ENV", value = "production" },
      { name = "DATABASE_URI", value = "postgresql://payload_admin:${var.db_password}@${aws_db_instance.postgres.endpoint}/${var.project_name}?sslmode=no-verify" },
      { name = "PAYLOAD_SECRET", value = var.payload_secret },
    ]
    logConfiguration = {
      logDriver = "awslogs"
      options = {
        "awslogs-group"         = "/ecs/${var.project_name}"
        "awslogs-region"        = var.aws_region
        "awslogs-stream-prefix" = "app"
        "awslogs-create-group"  = "true"
      }
    }
    healthCheck = {
      command     = ["CMD-SHELL", "wget -qO- http://localhost:3000/api/health || exit 1"]
      interval    = 30
      timeout     = 5
      retries     = 3
      startPeriod = 60
    }
  }])
}

resource "aws_iam_role" "ecs_execution" {
  name = "${var.project_name}-ecs-execution"

  assume_role_policy = jsonencode({
    Version = "2012-10-17"
    Statement = [{
      Action = "sts:AssumeRole"
      Effect = "Allow"
      Principal = { Service = "ecs-tasks.amazonaws.com" }
    }]
  })
}

resource "aws_iam_role_policy_attachment" "ecs_execution" {
  role       = aws_iam_role.ecs_execution.name
  policy_arn = "arn:aws:iam::aws:policy/service-role/AmazonECSTaskExecutionRolePolicy"
}

resource "aws_iam_role" "ecs_task" {
  name = "${var.project_name}-ecs-task"

  assume_role_policy = jsonencode({
    Version = "2012-10-17"
    Statement = [{
      Action = "sts:AssumeRole"
      Effect = "Allow"
      Principal = { Service = "ecs-tasks.amazonaws.com" }
    }]
  })
}

resource "aws_cloudwatch_log_group" "ecs" {
  name              = "/ecs/${var.project_name}"
  retention_in_days = 14
}
```

5. **infra/rds.tf** -- RDS PostgreSQL instance:

```hcl
resource "aws_db_instance" "postgres" {
  identifier     = "${var.project_name}-db"
  engine         = "postgres"
  engine_version = "16"
  instance_class = var.db_instance_class

  allocated_storage = 20
  storage_type      = "gp2"
  storage_encrypted = true

  db_name  = var.project_name
  username = "payload_admin"
  password = var.db_password

  publicly_accessible       = false
  skip_final_snapshot       = false
  final_snapshot_identifier = "${var.project_name}-db-final"

  backup_retention_period = 7
  backup_window           = "03:00-04:00"

  parameter_group_name = aws_db_parameter_group.postgres.name

  tags = {
    Name = "${var.project_name}-db"
  }
}

resource "aws_db_parameter_group" "postgres" {
  name   = "${var.project_name}-pg16"
  family = "postgres16"

  parameter {
    name  = "rds.force_ssl"
    value = "1"
  }
}
```

Note: `vpc_security_group_ids` and `db_subnet_group_name` are intentionally omitted because they depend on the user's VPC configuration. The user must add these before running `terraform apply`. A comment is included in the file explaining this.

Create `scripts/deploy.sh` with the following content:

```bash
#!/bin/bash
set -euo pipefail

# Configuration
AWS_REGION="${AWS_REGION:-ap-south-1}"
PROJECT_NAME="${PROJECT_NAME:-arifac}"
ECR_REPO="${PROJECT_NAME}-web"
ECS_CLUSTER="${PROJECT_NAME}-cluster"
ECS_SERVICE="${PROJECT_NAME}-web"

# Derived values
ACCOUNT_ID=$(aws sts get-caller-identity --query Account --output text)
ECR_URI="${ACCOUNT_ID}.dkr.ecr.${AWS_REGION}.amazonaws.com/${ECR_REPO}"

echo "=== ARIFAC Deployment ==="
echo "Region:  ${AWS_REGION}"
echo "Account: ${ACCOUNT_ID}"
echo "ECR:     ${ECR_URI}"
echo ""

# Validate required env vars
: "${PAYLOAD_SECRET:?PAYLOAD_SECRET is required}"
: "${NEXT_PUBLIC_SERVER_URL:?NEXT_PUBLIC_SERVER_URL is required}"

# Step 1: Login to ECR
echo ">>> Logging in to ECR..."
aws ecr get-login-password --region "$AWS_REGION" | docker login --username AWS --password-stdin "${ACCOUNT_ID}.dkr.ecr.${AWS_REGION}.amazonaws.com"

# Step 2: Build Docker image
echo ">>> Building Docker image..."
docker build \
  --build-arg NEXT_PUBLIC_SERVER_URL="$NEXT_PUBLIC_SERVER_URL" \
  --platform linux/amd64 \
  -t "$ECR_URI:latest" \
  -t "$ECR_URI:$(git rev-parse --short HEAD 2>/dev/null || echo 'dev')" \
  .

# Step 3: Push to ECR
echo ">>> Pushing to ECR..."
docker push "$ECR_URI:latest"
docker push "$ECR_URI:$(git rev-parse --short HEAD 2>/dev/null || echo 'dev')"

# Step 4: Force new ECS deployment
echo ">>> Updating ECS service..."
aws ecs update-service \
  --cluster "$ECS_CLUSTER" \
  --service "$ECS_SERVICE" \
  --force-new-deployment \
  --region "$AWS_REGION" \
  --no-cli-pager

echo ""
echo "=== Deployment initiated ==="
echo "Monitor at: https://${AWS_REGION}.console.aws.amazon.com/ecs/v2/clusters/${ECS_CLUSTER}/services/${ECS_SERVICE}"
echo ""
echo "NOTE: Run database migrations separately if schema has changed:"
echo "  DATABASE_URI=<production_uri> pnpm payload migrate"
```

Make the script executable.

Key design decisions:
- Does NOT bake DATABASE_URI into the image (runtime env var via ECS task definition)
- Tags images with both `latest` and git short SHA for traceability
- Uses `--platform linux/amd64` to ensure compatibility with Fargate (important when building on Apple Silicon)
- Migration is a separate manual step (not automated in deploy) because it should be verified before updating the service
- Validates required env vars before starting

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Terraform state to secrets | terraform.tfstate contains db_password and payload_secret in plaintext |
| Deploy script to AWS | Script authenticates via AWS CLI credentials and pushes images |
| ECS task to RDS | Database credentials passed as plaintext env vars in task definition |
| Internet to ECS ALB | Public traffic reaches the application through ALB |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-04-06 | Information Disclosure | infra/terraform.tfstate | mitigate | Local backend used (not committed to git); terraform.tfvars excluded from git via .gitignore; user advised to migrate to S3 backend with encryption for production |
| T-04-07 | Information Disclosure | infra/terraform.tfvars | mitigate | .tfvars contains secrets; terraform.tfvars.example has CHANGE_ME placeholders; real .tfvars must NOT be committed |
| T-04-08 | Information Disclosure | ECS task definition | accept | DATABASE_URI and PAYLOAD_SECRET in plaintext env vars; acceptable for v1 with 1-2 editors; recommend Secrets Manager for v2 |
| T-04-09 | Tampering | RDS | mitigate | rds.force_ssl = 1 parameter enforces SSL on all connections; storage_encrypted = true for at-rest encryption |
| T-04-10 | Denial of Service | RDS | mitigate | publicly_accessible = false prevents direct internet access to database |
| T-04-11 | Spoofing | deploy.sh | mitigate | Script validates AWS identity via `aws sts get-caller-identity`; ECR login uses IAM-based authentication |
</threat_model>
